			is_greater_power = yes
			is_secondary_power = yes

		}
		is_disarmed = no

		THIS = {
			is_greater_power = yes
			is_disarmed = no
			mass_politics = 1
			has_country_flag != in_great_war
		}
	}

	po_disarmament = yes
	po_reparations = yes

	war_name = WAR_DISMANTLE_NAME

	on_add = {
		FROM = {
			set_country_flag = dismantle_declared
		}
		move_issue_percentage = {
			from = pro_military
			to = jingoism
			value = 0.10
		}
		country_event = 96000
	}

	on_po_accepted = {
		set_country_flag = dismantling_treaty
	}
}
